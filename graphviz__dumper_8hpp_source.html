14// Helper function for escaping special characters in DOT
15static std::string escape_dot(const std::string &s) {
16 std::string out;
17 out.reserve(s.size());
18 for (char c : s) {
19 if (c == '"' || c == '\\' || c == '{' || c == '}' || c == '|' ||
20 c == '<' || c == '>') {
21 out += '\\';
22 }
23 out += c;
24 }
25 return out;
26}

28static std::string escape_html(const std::string &s) {
29 std::string out;
30 out.reserve(s.size());
31 for (char c : s) {
32 switch (c) {
33 case '&':
34 out += "&amp;";
35 break;
36 case '<':
37 out += "&lt;";
38 break;
39 case '>':
40 out += "&gt;";
41 break;
42 default:
43 out += c;
44 break;
45 }
46 }
47 return out;
48}

249 static void dump_attributes(const Attributes &attrs, std::string &label) {
250 for (const auto &[attr_name, attr] : attrs) {
251 std::string attr_str = escape_html(attr_name) + "=";
252 std::visit(
253 [&attr_str](const auto &val) {
254 using T = std::decay_t<decltype(val)>;
255 if constexpr (std::is_same_v<T, float>) {
256 attr_str += std::to_string(val);
257 } else if constexpr (std::is_same_v<T, int64_t>) {
258 attr_str += std::to_string(val);
259 } else if constexpr (std::is_same_v<T, std::string>) {
260 attr_str += "\"" + escape_html(val) + "\"";
261 } else if constexpr (std::is_same_v<T,
262 std::vector<float>>) {
263 attr_str += "[";
264 for (size_t i = 0; i < val.size(); ++i) {
265 if (i > 0)
266 attr_str += ",";
267 attr_str += std::to_string(val[i]);
268 }
269 attr_str += "]";
270 } else if constexpr (std::is_same_v<T,
271 std::vector<int64_t>>) {
272 attr_str += "[";
273 for (size_t i = 0; i < val.size(); ++i) {
274 if (i > 0)
275 attr_str += ",";
276 attr_str += std::to_string(val[i]);
277 }
278 attr_str += "]";
279 }
280 },
281 attr.get_value());
282 label += "<tr><td align=\"left\">" + attr_str + "</td></tr>";
