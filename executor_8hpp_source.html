76inline std::vector<const Node *> Executor::topological_sort() {
77 std::unordered_map<std::string, const Node *> producer_of;
78 // init producer_of ---------------------------------------------
79 for (const auto &node : graph_.get_nodes()) {
80 for (const auto &out : node.get_outputs()) {
81 producer_of[out] = &node;
82 }
83 }
84 // --------------------------------------------------------------
85
86 std::unordered_map<const Node *, std::vector<const Node *>> adj;
87 std::unordered_map<const Node *, std::size_t> in_degree;
88 // init adj and in_degree ---------------------------------------
89 for (const auto &node : graph_.get_nodes()) {
90 const Node *p = &node;
91 adj[p] = {};
92 in_degree[p] = 0;
93 }
94 // --------------------------------------------------------------
95
96 // To avoid duplicate edges
97 std::unordered_set<std::pair<const Node *, const Node *>, Edge_hash> seen;
98
99 // build edges from inputs --------------------------------------
100 for (const auto &consumer_node : graph_.get_nodes()) {
101 const Node *consumer = &consumer_node;
102
103 for (const auto &input_tensor : consumer_node.get_inputs()) {
104 auto it = producer_of.find(input_tensor);
105 if (it == producer_of.end()) {
106 continue;
107 }
108 const Node *producer = it->second;
109 std::pair<const Node *, const Node *> edge{producer, consumer};
110 if (seen.insert(edge).second) {
111 adj[producer].push_back(consumer);
112 in_degree[consumer] += 1;
113 }
114 }
115 }
116 // --------------------------------------------------------------
117
118 std::deque<const Node *> kahn_queue;
119 // init kahn_queue ----------------------------------------------
120 for (const auto &node : graph_.get_nodes()) {
121 const Node *p = &node;
122 if (in_degree[p] == 0)
123 kahn_queue.push_back(p);
124 }
125 // --------------------------------------------------------------
126
127 const std::size_t num_of_nodes = graph_.get_nodes().size();
128
129 std::vector<const Node *> ready_nodes;
130 ready_nodes.reserve(num_of_nodes);
131
132 // add ready nodes to vector -----------------------------------
133 while (!kahn_queue.empty()) {
134 const Node *ready_node = kahn_queue.front();
135 kahn_queue.pop_front();
136 ready_nodes.push_back(ready_node);
137
138 for (const Node *output_tensors : adj[ready_node]) {
139 if (--in_degree[output_tensors] == 0) {
140 kahn_queue.push_back(output_tensors);
141 }
142 }
143 }
144 // --------------------------------------------------------------
145
146 if (ready_nodes.size() != num_of_nodes) {
147 throw std::runtime_error("Topological sort failed: "
148 "graph has a cycle (or unresolved deps).");
149 }
150
151 return ready_nodes;
152}
