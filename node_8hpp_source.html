185 for (const auto &attr : node.attribute()) {
186 const std::string &name = attr.name();
187
188 switch (attr.type()) {
189 case onnx::AttributeProto_AttributeType_FLOAT: {
190 set_attribute(name, attr.f());
191 break;
192 }
193
194 case onnx::AttributeProto_AttributeType_INT: {
195 set_attribute(name, static_cast<int64_t>(attr.i()));
196 break;
197 }
198
199 case onnx::AttributeProto_AttributeType_STRING: {
200 set_attribute(name, attr.s());
201 break;
202 }
203
